// Add a notification and make it automatically close in 20 seconds
App.notification.add("Hello World", "You have got a new message from John Tomason. Check it out.");

// Add a notification and make it automatically close in 20 seconds
App.notification.add("Hello World", "You have got a new message from John Tomason. Check it out.", 20000);

// remove all notifications
App.notification.removeAll();

// remove the first notification
App.notification.removeFirst();

// remove the last notification
App.notification.removeLast();

// remove the nth notification, index starting from 0
App.notification.removeAt(1);
